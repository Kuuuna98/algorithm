import java.io.BufferedReader;
import java.io.InputStreamReader;

public class Main_1107_ë¦¬ëª¨ì»¨ {
    static int answer;
    static boolean[] brokenRemote;

    public static void main(String args[]) throws Exception {
        int N = init();
        int tmp = checkCount(N);
        if (tmp > 0 && tmp < answer) answer = tmp;

        for (int i = 1; i <= N; i++) {
            tmp = checkCount(N - i);
            if (tmp > 0) {
                if (tmp + i < answer) {
                    answer = tmp + i;
                } else {
                    break;
                }
            }
        }

        for (int i = 1; i <= answer; i++) {
            tmp = checkCount(N + i);
            if (tmp > 0) {
                if (tmp + i < answer) {
                    answer = tmp + i;
                } else {
                    break;
                }
            }
        }

        System.out.println(answer);
    }

    static int init() throws Exception {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        int N = Integer.parseInt(br.readLine());
        answer = Math.abs(N - 100);
        brokenRemote = new boolean[10];
        int M = Integer.parseInt(br.readLine());
        if (M > 0) {
            String input = br.readLine();
            for (int m = 0; m < M; m++) {
                brokenRemote[input.charAt(m + m) - '0'] = true;
            }
        }
        return N;
    }

    static int checkCount(int number) {
        int cnt = 1;
        if (brokenRemote[number % 10]) return -1;
        while ((number /= 10) > 0) {
            if (brokenRemote[number % 10]) return -1;
            cnt++;
        }
        return cnt;
    }
}
